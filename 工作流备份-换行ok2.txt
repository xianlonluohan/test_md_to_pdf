name: "Markdown to PDF Converter"

on:
  push:
    branches: ["main", "master"]
    tags: ["v*", "pdf-*"]
  pull_request:
    branches: ["main", "master"]
  workflow_dispatch:

permissions:
  contents: write
  pages: write
  id-token: write

jobs:
  convert-md-to-pdf:
    runs-on: ubuntu-latest
    
    steps:
      - name: "Checkout repository"
        uses: actions/checkout@v4
        with:
          fetch-depth: 0
      
      - name: "Install dependencies"
        run: |
          sudo apt-get update
          sudo apt-get install -y \
            pandoc \
            texlive-xetex \
            texlive-latex-extra \
            texlive-fonts-recommended \
            texlive-latex-recommended \
            texlive-lang-chinese \
            fonts-noto-cjk \
            fonts-noto-cjk-extra \
            fonts-wqy-microhei \
            fonts-wqy-zenhei \
            lmodern \
            tree \
            jq
      
      - name: "Create LaTeX template"
        run: |
          # 创建模板，移除换行箭头
          cat > header.tex << 'EOF'
          % 中文支持 - 使用简单可靠的设置
          \usepackage{xeCJK}
          \usepackage[UTF8]{ctex}
          
          % 简单设置中文字体，让系统自动查找
          \setCJKmainfont{Noto Serif CJK SC}
          \setCJKsansfont{Noto Sans CJK SC}
          \setCJKmonofont{Noto Sans Mono CJK SC}
          
          % 页面设置 - 使用中等边距
          \usepackage[margin=1.2in]{geometry}
          
          % 超链接
          \usepackage{hyperref}
          \hypersetup{
            colorlinks=true,
            linkcolor=blue,
            urlcolor=blue,
            citecolor=blue,
            pdfborder={0 0 0},
          }
          
          % 代码块设置 - 使用改进的 listings
          \usepackage{listings}
          \usepackage{xcolor}
          
          % 定义颜色
          \definecolor{codebg}{rgb}{0.98,0.98,0.98}
          \definecolor{framecolor}{rgb}{0.8,0.8,0.8}
          \definecolor{keywordcolor}{rgb}{0,0,1}
          \definecolor{commentcolor}{rgb}{0,0.5,0}
          \definecolor{stringcolor}{rgb}{0.64,0.08,0.08}
          
          % 改进的代码块设置 - 移除换行箭头
          \lstset{
            basicstyle=\ttfamily\footnotesize,
            backgroundcolor=\color{codebg},
            frame=single,
            framesep=6pt,
            framerule=0.8pt,
            rulecolor=\color{framecolor},
            breaklines=true,
            breakatwhitespace=false,  % 允许在任何字符处换行
            breakautoindent=true,
            breakindent=0pt,
            % 移除换行箭头，改为空或空格
            postbreak=\mbox{},
            showstringspaces=false,
            tabsize=2,
            numbers=none,
            xleftmargin=12pt,
            xrightmargin=12pt,
            upquote=true,  % 正确处理引号
            columns=fullflexible,  % 更好的字符间距
            keepspaces=true,  % 保持空格
            keywordstyle=\color{keywordcolor}\bfseries,
            commentstyle=\color{commentcolor}\itshape,
            stringstyle=\color{stringcolor},
          }
          
          % 处理过长的等号和连字符
          \usepackage{seqsplit}
          
          % 段落设置
          \usepackage{parskip}
          \setlength{\parindent}{0pt}
          \setlength{\parskip}{0.8em}
          
          % 防止溢出
          \sloppy
          \emergencystretch=3em
          EOF
      
      - name: "Set up output directory"
        run: |
          mkdir -p pdf_output
      
      - name: "Convert Markdown to PDF"
        run: |
          echo "开始转换Markdown文件到PDF..."
          
          # 查找所有md文件
          MD_FILES=$(find . -name "*.md" -not -path "./.*" -not -path "*/node_modules/*" | sed 's/^\.\///')
          
          echo "找到的文件数量: $(echo "$MD_FILES" | wc -l)"
          
          IFS=$'\n'
          converted_count=0
          failed_count=0
          failed_files=""
          
          for md_file in $MD_FILES; do
            [ -z "$md_file" ] && continue
            
            dir_name=$(dirname "$md_file")
            base_name=$(basename "$md_file" .md)
            
            # 创建输出目录
            mkdir -p "pdf_output/$dir_name"
            
            echo "正在转换: $md_file"
            
            # 获取Markdown文件所在目录
            md_dir=$(dirname "$md_file")
            
            # 复制资源文件
            if [ -d "$dir_name/picture" ]; then
              mkdir -p "pdf_output/$dir_name/picture"
              find "$dir_name/picture" -type f \( -name "*.png" -o -name "*.jpg" -o -name "*.jpeg" \) -exec cp {} "pdf_output/$dir_name/picture/" 2>/dev/null \; || true
            fi
            
            if [ -d "$dir_name/resource" ]; then
              mkdir -p "pdf_output/$dir_name/resource"
              find "$dir_name/resource" -type f -exec cp {} "pdf_output/$dir_name/resource/" 2>/dev/null \; || true
            fi
            
            # 转换Markdown为PDF
            set +e
            
            echo "使用简化方法转换..."
            
            # 简化转换方法，避免复杂的字体设置
            pandoc "$md_file" \
              --resource-path="$md_dir:.:." \
              -o "pdf_output/$dir_name/$base_name.pdf" \
              --pdf-engine=xelatex \
              --listings \
              --include-in-header=header.tex \
              -V mainfont="Noto Serif CJK SC" \
              -V sansfont="Noto Sans CJK SC" \
              -V monofont="Noto Sans Mono CJK SC" \
              -V geometry:margin=1.2in \
              -V geometry:a4paper \
              -V colorlinks=true \
              -V linkcolor=blue \
              -V urlcolor=blue \
              -V lang=zh-CN \
              -V papersize=a4 \
              --wrap=auto \
              --highlight-style=tango \
              2>&1 | tee pandoc.log
            
            set -e
            
            # 检查是否生成PDF
            if [ -f "pdf_output/$dir_name/$base_name.pdf" ] && [ -s "pdf_output/$dir_name/$base_name.pdf" ]; then
              file_size=$(du -h "pdf_output/$dir_name/$base_name.pdf" | cut -f1)
              
              # 检查文件头部是否有效
              if file "pdf_output/$dir_name/$base_name.pdf" | grep -q "PDF document"; then
                echo "  ✓ 成功: $base_name.pdf ($file_size)"
                converted_count=$((converted_count + 1))
              else
                echo "  ⚠ 可能有问题: $base_name.pdf ($file_size)"
                failed_count=$((failed_count + 1))
                failed_files="$failed_files\n  - $md_file"
              fi
            else
              echo "  ✗ 失败: $md_file"
              failed_count=$((failed_count + 1))
              failed_files="$failed_files\n  - $md_file"
            fi
          done
          
          echo ""
          echo "================================"
          echo "转换完成!"
          echo "成功: $converted_count 个文件"
          echo "失败: $failed_count 个文件"
          
          if [ $failed_count -gt 0 ]; then
            echo ""
            echo "失败的文件:"
            echo -e "$failed_files"
          fi
          
          echo "================================"
          
          # 如果所有文件都失败，则退出
          if [ $converted_count -eq 0 ] && [ $failed_count -gt 0 ]; then
            echo "错误: 没有成功转换任何文件"
            exit 1
          fi
      
      - name: "Verify PDF files"
        run: |
          echo "验证PDF文件..."
          
          # 安装pdfinfo用于验证PDF
          sudo apt-get install -y poppler-utils
          
          total_pdfs=$(find pdf_output -name "*.pdf" -type f 2>/dev/null | wc -l)
          
          if [ $total_pdfs -eq 0 ]; then
            echo "错误: 没有找到任何PDF文件"
            exit 1
          fi
          
          echo "总计生成 $total_pdfs 个PDF文件"
          
          # 验证每个PDF文件
          valid_count=0
          invalid_files=""
          
          find pdf_output -name "*.pdf" -type f | sort | while read pdf_file; do
            size=$(du -h "$pdf_file" | cut -f1)
            
            # 使用pdfinfo验证PDF
            if pdfinfo "$pdf_file" >/dev/null 2>&1; then
              echo "  ✓ 有效: $pdf_file ($size)"
              valid_count=$((valid_count + 1))
            else
              echo "  ✗ 无效: $pdf_file ($size)"
              invalid_files="$invalid_files\n  - $pdf_file"
            fi
          done
          
          echo ""
          echo "有效PDF文件: $valid_count"
          
          if [ -n "$invalid_files" ]; then
            echo "无效的PDF文件:"
            echo -e "$invalid_files"
          fi
      
      - name: "Create release documentation"
        run: |
          echo "创建发布文档..."
          
          # 获取 tag 名称（如果有）
          if [[ "${{ github.ref }}" == refs/tags/* ]]; then
            TAG_NAME="${GITHUB_REF#refs/tags/}"
          else
            TAG_NAME="latest"
          fi
          
          # 创建 README
          echo "# PDF 文档集" > pdf_output/README.md
          echo "" >> pdf_output/README.md
          echo "## 生成信息" >> pdf_output/README.md
          echo "- **生成时间**: $(date '+%Y-%m-%d %H:%M:%S')" >> pdf_output/README.md
          echo "- **GitHub SHA**: ${{ github.sha }}" >> pdf_output/README.md
          echo "- **触发事件**: ${{ github.event_name }}" >> pdf_output/README.md
          if [[ "${{ github.ref }}" == refs/tags/* ]]; then
            echo "- **标签**: $TAG_NAME" >> pdf_output/README.md
          fi
          echo "- **代码块样式**: 使用带边框的代码框，自动换行" >> pdf_output/README.md
          echo "- **页面边距**: 1.2英寸边距，防止内容溢出" >> pdf_output/README.md
          echo "- **中文字体**: 使用Noto CJK字体" >> pdf_output/README.md
          echo "- **改进功能**: 移除换行箭头符号，优化代码显示" >> pdf_output/README.md
          echo "" >> pdf_output/README.md
          
          echo "## 文件列表" >> pdf_output/README.md
          echo "" >> pdf_output/README.md
          
          # 统计文件数量
          PDF_COUNT=$(find pdf_output -name "*.pdf" -type f 2>/dev/null | wc -l)
          
          # 按目录组织文件列表
          current_dir=""
          find pdf_output -name "*.pdf" -type f | sort | while read pdf_file; do
            rel_path=${pdf_file#pdf_output/}
            dir_name=$(dirname "$rel_path")
            file_name=$(basename "$rel_path")
            file_size=$(du -h "$pdf_file" | cut -f1)
            
            # 如果是新的目录，添加标题
            if [ "$current_dir" != "$dir_name" ]; then
              current_dir="$dir_name"
              if [ "$dir_name" != "." ]; then
                echo "### $dir_name/" >> pdf_output/README.md
                echo "" >> pdf_output/README.md
              fi
            fi
            
            echo "- [$file_name]($rel_path) ($file_size)" >> pdf_output/README.md
          done
          
          echo "" >> pdf_output/README.md
          echo "**总计: $PDF_COUNT 个 PDF 文件**" >> pdf_output/README.md
      
      - name: "Create zip archive for release"
        if: startsWith(github.ref, 'refs/tags/')
        run: |
          echo "为 GitHub Releases 创建压缩包..."
          
          # 获取 tag 名称
          TAG_NAME="${GITHUB_REF#refs/tags/}"
          
          # 创建压缩包
          cd pdf_output
          zip -r "../pdf-documents-$TAG_NAME.zip" .
          cd ..
          
          echo "压缩包已创建: pdf-documents-$TAG_NAME.zip"
          ls -lh pdf-documents-*.zip
      
      - name: "Upload PDF artifacts"
        uses: actions/upload-artifact@v4
        with:
          name: pdf-documents
          path: |
            pdf_output/
            pdf-documents-*.zip
          retention-days: 7
      
      - name: "Create GitHub Release"
        if: startsWith(github.ref, 'refs/tags/')
        id: create_release
        uses: softprops/action-gh-release@v1
        with:
          tag_name: ${{ github.ref_name }}
          name: "PDF Documents - ${{ github.ref_name }}"
          body_path: pdf_output/README.md
          files: |
            pdf-documents-${{ github.ref_name }}.zip
          draft: false
          prerelease: false
          generate_release_notes: true
      
      - name: "Verify release creation"
        if: startsWith(github.ref, 'refs/tags/')
        run: |
          echo "Release 创建完成!"
          echo "访问: https://github.com/${{ github.repository }}/releases/tag/${{ github.ref_name }}"
